
import openai
import os

openai.api_key = os.getenv("")

while True:
    prompt = input("You: ")
    if prompt.lower() in ["exit", "quit"]:
        break

    try:
        response = openai.ChatCompletion.create(
            model="gpt-5.1",
            messages=[
                {"role": "user", "content": prompt}
            ]
        )

        ai_text = response['choices'][0]['message']['content']
        print("AI:", ai_text)

    except Exception as e:
        print("Error:", e)
